    **Step 8: Build post-2000 test matrix in annual chunks, write incrementally**
    ```python
    import pyarrow as pa
    import pyarrow.parquet as pq

    test_path = Path('data/processed/feature_matrix_test.parquet')
    writer = None
    for year in tqdm(range(2000, 2027), desc="Building test set"):
        year_key = str(year)
        if year_key not in ephe_encoded.index.year.astype(str).unique():
            continue  # skip years not in ephemeris (e.g. partial 2026)
        year_ephe = ephe_encoded[ephe_encoded.index.year == year]
        year_chunks = []
        for date_val, row in year_ephe.iterrows():
            chunk = build_matrix_chunk(row, active_cells, eq_index, country_map)
            year_chunks.append(chunk)
        year_df = pd.concat(year_chunks, ignore_index=True)

        table = pa.Table.from_pandas(year_df, preserve_index=False)
        if writer is None:
            writer = pq.ParquetWriter(str(test_path), table.schema, compression='snappy')
        writer.write_table(table)
        del year_df, year_chunks
    if writer:
        writer.close()
    logger.info(f"Wrote {test_path}")
    ```
